<script setup>
import { ref, computed } from 'vue'
import FilterButton from './FilterButton.vue'
import MonthSection from './MonthSection.vue'
import { timelineData } from '../data/timelineData.js'

// Initialize with the imported data
const data = ref(timelineData.timeline)

// Active filter state
const activeFilter = ref('all')

// Filter handling
const setFilter = (filter) => {
  activeFilter.value = filter
}

// Computed filtered data
const filteredData = computed(() => {
  return data.value.map(monthGroup => ({
    ...monthGroup,
    entries: monthGroup.entries.filter(entry => {
      if (activeFilter.value === 'all') return true
      return entry.type === activeFilter.value
    })
  }))
})
</script>

<template>
  <div class="timeline-container">
    <!-- Filter buttons -->
    <div class="toggle-container">
      <FilterButton
        label="Show All"
        :active="activeFilter === 'all'"
        @click="() => setFilter('all')"
        color="blue"
      />
      <!-- Other filter buttons... -->
    </div>

    <!-- Timeline content -->
    <div class="timeline">
      <MonthSection
        v-for="(monthGroup, index) in filteredData"
        :key="index"
        :month="monthGroup.month"
        :entries="monthGroup.entries"
      />
    </div>
  </div>
</template>

<script setup>
/**
 * TimelineCard Component
 * Displays a single AI model release event with type-based styling
 * 
 * Props:
 * @param {string} title - Name of the AI model
 * @param {string} description - Description of the model release
 * @param {string} type - Type of release (api, weights, german)
 * @param {string} [link] - Optional link to the model
 */
const props = defineProps({
  title: String,
  description: String,
  type: String,
  link: {
    type: String,
    default: null
  }
});

// Compute styling based on model type
const getTypeStyles = () => {
  switch (props.type) {
    case 'api':
      return {
        border: 'border-l-4 border-red-500',
        badge: {
          classes: 'tag api-only',
          text: 'API Only'
        }
      };
    case 'weights':
      return {
        border: 'border-l-4 border-green-500',
        badge: {
          classes: 'tag open-source',
          text: 'Open Weights'
        }
      };
    case 'german':
      return {
        border: 'border-l-4 border-purple-500',
        badge: {
          classes: 'tag german',
          text: 'German Model'
        }
      };
    default:
      return {
        border: '',
        badge: null
      };
  }
};
</script>

<template>
  <div 
    class="event" 
    :class="getTypeStyles().border"
  >
    <h4>{{ title }}</h4>
    <p>
      {{ description }}
      <a v-if="link" :href="link" target="_blank" rel="noopener">Learn more</a>
    </p>
    <span 
      v-if="getTypeStyles().badge"
      :class="getTypeStyles().badge.classes"
    >
      {{ getTypeStyles().badge.text }}
    </span>
  </div>
</template>
